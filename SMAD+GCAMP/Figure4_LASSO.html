LASSO Regression for AUC

library(glmnet)
library(clusterProfiler)
library(org.Hs.eg.db)
library(ggplot2)

Load Data and fit LASSO model

expression <- read.csv("filtered_expression_V4.csv", row.names = 1)
cell_metadata <- read.csv("cell_metadata_V4.csv", row.names = 1)
X <- as.matrix(expression)
y <- cell_metadata$AUC_above_median_log
cv_fit <- cv.glmnet(X, y, alpha = 1, family = "gaussian")
plot(cv_fit)

Load the best model

coef_matrix <- as.matrix(coef(cv_fit, s = "lambda.min"))
lasso_genes <- rownames(coef_matrix)[coef_matrix != 0]
pos_lasso_genes <- rownames(coef_matrix)[coef_matrix > 0]
cat("Genes in best model:", length(lasso_genes)
,"\nGenes with positive coefs in best model:", length(pos_lasso_genes), "\n")
## Genes in best model: 978 
## Genes with positive coefs in best model: 385

Prediction accuracy

y_pred <- predict(cv_fit, newx = X, s = "lambda.min")
mse_train <- mean((y - y_pred)^2)
print(paste("Training Mean Squared Error(MSE) of best model:", mse_train))
## [1] "Training Mean Squared Error(MSE) of best model: 0.4114826812364"
ss_total_train <- sum((y - mean(y))^2)
ss_residual_train <- sum((y - y_pred)^2)
r_squared_train <- 1 - (ss_residual_train / ss_total_train)
print(paste("Training R-Squared(R^2) of best model:", r_squared_train))
## [1] "Training R-Squared(R^2) of best model: 0.216438381074261"
spearman_corr <- cor(y, y_pred, method = "spearman", use = "complete.obs")
print(paste("Spearman's rank correlation of best model:", spearman_corr))
## [1] "Spearman's rank correlation of best model: 0.407417533092328"

GO analysis of the genes with positive coefs in best model

background_genes <- readLines("background_genes.txt")
go_results <- enrichGO(
  gene = pos_lasso_genes,       # List of significant genes
  OrgDb = org.Hs.eg.db,           # Human genome annotation
  keyType = "SYMBOL",             # Using human gene symbols
  ont = "BP",                     # Ontology: BP (Biological Process)
  pAdjustMethod = "BH",           # Multiple testing correction method
  pvalueCutoff = 0.05,            # P-value cutoff for significance
  qvalueCutoff = 0.05,           # Q-value cutoff for significance
  universe = background_genes     # Optional: Background gene set
)
dotplot(go_results, showCategory = 10, title = "Top 10 GO Terms from LASSO Regression") +
  theme(panel.grid.major = element_blank(),
        panel.grid.minor = element_blank())
